2. Cart & Checkout

		
class Cart:
    def __init__(self):   
        self.products = []   
    def add(self, obj):   
        self.products.append(obj)   
    def notify(self):
        for obj in self.products:   
            obj.notify(len(self.products))
    def price(self):
        for obj in self.products:   
            print(obj.price)
