例 2.7. Functions with parameters sample

   exit
}

    echo $1
}
e Hello

quit

5.1. Local variables

 #!/bin/bash
	HELLO=Hello
	function hello {
	        local HELLO=World
	        echo $HELLO
	}
	echo $HELLO
	hello
	echo $HELLO
