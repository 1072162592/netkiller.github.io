2. case

例 11.3. case

case "$1" in
        start)
            start
            ;;
        stop)
            stop
            ;;
        status)
            status 
            ;;
        restart)
            stop
            start
            ;;
        condrestart)
            condrestart
            ;;
         
        *)
            echo $"Usage: $0 {start|stop|restart|condrestart|status}"
            exit 1
esac
