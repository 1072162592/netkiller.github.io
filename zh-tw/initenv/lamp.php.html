<IfModule mpm_worker_module>
        StartServers        //Apache啟動立即產生n個子進程
        MaxClients          //允許最大的客戶數
        ServerLimit         //最大允許n子進程數
        MinSpareThreads     //最少有n個空綫程
        MaxSpareThreads     //最多有n個空綫程
        ThreadLimit         //最多有n個綫程
        ThreadsPerChild     //一個子進程有n個常駐綫程
        MaxRequestsPerChild //當處理的綫程數達到n銷毀
</IfModule>

<IfModule mpm_worker_module>

</IfModule>
