5. xinetd - replacement for inetd with many enhancements

$ sudo apt-get install xinetd

5.1. tftpd

apt-get install xinetd
apt-get install tftpd tftp

/etc/xinetd.d/tftp

service tftp
{
	disable=no
	socket_type=dgram
	protocol =udp
	wait=yes
	user=root
	server=/usr/sbin/in.tftpd
	server_args =-s /home/neo/tftpboot -c
	per_source=11
	cps=100 2
	flags=IPv4
}
