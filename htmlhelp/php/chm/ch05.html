<?php

$extensions = get_loaded_extensions();
foreach($extensions as $each_ext)
{
	echo "\t<keyword>$each_ext, ";
	$ext_funcs = get_extension_funcs($each_ext);
	foreach($ext_funcs as $func)
	{
	   echo "$func, ";
	}
	echo "</keyword>\r\n";
}
?>
