2.2. php://stdin & php://stdout

		
$ cat test.php
#!/usr/bin/php
<?php

$handle = fopen("php://stdin", "r");

while (!feof($handle)) {
    $buffer = fgets($handle);
    print($buffer);
}
fclose($handle);
		
		
$ cat /etc/passwd | ./test.php
