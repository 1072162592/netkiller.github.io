xcache.count = 24
xcache.gc_interval = 300
xcache.var_size  = 16M
xcache.var_count = 24
xcache.var_gc_interval = 300
xcache.readonly_protection = On
xcache.coredump_directory =   ""
xcache.optimizer = On

[xcache.coverager]

xcache.coveragedump_directory = ""
EOF
